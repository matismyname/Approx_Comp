def remove_noise(s):

    td = 55               #Threshold for comparing the rgb values of nearby cells
    im = Image.open(s)
    pix = im.load()
    
    width = im.size[0]
    height = im.size[1]
    
    #Ignoring the borders
    for h in range(height - 1):
        for w in range(width - 1):
            
            #If the rgb values of the nearby cells are larger than
            #the ones compared to, assign the rgb values of a nearby cell to the pixel with the large rgb values 
            if(h != height - 1):
                if(w != width - 1):
                    if(np.abs(pix[w, h][0] - pix[w + 1, h][0]) > td and
                       np.abs(pix[w, h][0] - pix[w, h + 1][0]) > td and
                       np.abs(pix[w, h][0] - pix[w + 1, h + 1][0]) > td):
                        pix[w, h] = pix[w+1, h]
                    
                    elif(np.abs(pix[w, h][1] - pix[w + 1, h][1]) > td and
                       np.abs(pix[w, h][1] - pix[w, h + 1][1]) > td and
                        np.abs(pix[w, h][1] - pix[w + 1, h + 1][1]) > td):
                        pix[w, h] = pix[w+1, h]
        

                    elif(np.abs(pix[w, h][2] - pix[w + 1, h][2]) > td and
                       np.abs(pix[w, h][2] - pix[w, h + 1][2]) > td and
                        np.abs(pix[w, h][2] - pix[w + 1, h + 1][2]) > td):
                        pix[w, h] = pix[w+1, h]

    return im

#Calculate the PSNR
def calc_psnr(org, con):
    img1 = cv2.imread(org)
    img2 = cv2.imread(con)
    psnr = cv2.PSNR(img1, img2)
    
    return psnr

#Implementation of the median filter
#3x3 window
def remove_noise_median(s):
    
    im = Image.open(s)
    pix = im.load()
    
    width = im.size[0]
    height = im.size[1]
    
    for h in range(0, height-2, 3):
        for w in range(0, width-2, 3):
            r_s = [
                pix[w, h][0], pix[w+1, h][0], pix[w+2, h][0],
                pix[w, h+1][0], pix[w+1, h+1][0], pix[w+2, h+1][0],
                pix[w, h+2][0], pix[w+1, h+2][0], pix[w+2, h+2][0]
            ]
            g_s = [
                pix[w, h][1], pix[w+1, h][1], pix[w+2, h][1],
                pix[w, h+1][1], pix[w+1, h+1][1], pix[w+2, h+1][1],
                pix[w, h+2][1], pix[w+1, h+2][1], pix[w+2, h+2][1]
            ]
            b_s = [
                pix[w, h][2], pix[w+1, h][2], pix[w+2, h][2],
                pix[w, h+1][2], pix[w+1, h+1][2], pix[w+2, h+1][2],
                pix[w, h+2][2], pix[w+1, h+2][2], pix[w+2, h+2][2]
            ]
            
            
            r_s = st.median(r_s)
            g_s = st.median(g_s)
            b_s = st.median(b_s)
            
            
            pix[w, h] = (r_s, g_s, b_s)
            pix[w+1, h] = (r_s, g_s, b_s)
            pix[w+2, h] = (r_s, g_s, b_s)
            
            pix[w, h+1] = (r_s, g_s, b_s)
            pix[w+1, h+1] = (r_s, g_s, b_s)
            pix[w+2, h+1] = (r_s, g_s, b_s)
            
            pix[w, h+2] = (r_s, g_s, b_s)
            pix[w+1, h+2] = (r_s, g_s, b_s)
            pix[w+2, h+2] = (r_s, g_s, b_s)

    return im
